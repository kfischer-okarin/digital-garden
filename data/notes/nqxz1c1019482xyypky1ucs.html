Logic Operators

And

Inside the test expression

if [ "$1" = "run" -a "$2" != "" ]; then
  # With run and an additional argument
fi

Connecting test expressions

if [ "$1" = "run" ] && [ "$2" != "" ]; then
  # With run and an additional argument
fi

Or

Inside the test expression

if [ "$1" = "-h" -o "$1" = "--help" ]; then
  # When the argument is -h or --help
fi

Connecting test expressions

if [ "$1" = "-h" ] || [ "$1" = "--help" ]; then
  # When the argument is -h or --help
fi
